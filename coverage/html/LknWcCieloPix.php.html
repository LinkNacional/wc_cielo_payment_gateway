577    public function validateCpfCnpj($cpfCnpj)
578    {
579        // Remove caracteres especiais
580        $cpfCnpj = preg_replace('/[^0-9]/', '', $cpfCnpj);
581
582        // Verifica se é CPF
583        if (strlen($cpfCnpj) === 11) {
584            // Verifica se todos os dígitos são iguais
585            if (preg_match('/(\d)\1{10}/', $cpfCnpj)) {
586                return false;
587            }
588
589            // Calcula o primeiro dígito verificador
590            $sum = 0;
591            for ($i = 0; $i < 9; $i++) {
592                $sum += (int) ($cpfCnpj[$i]) * (10 - $i);
593            }
594            $digit1 = ($sum % 11 < 2) ? 0 : 11 - ($sum % 11);
595
596            // Calcula o segundo dígito verificador
597            $sum = 0;
598            for ($i = 0; $i < 10; $i++) {
599                $sum += (int) ($cpfCnpj[$i]) * (11 - $i);
600            }
601            $digit2 = ($sum % 11 < 2) ? 0 : 11 - ($sum % 11);
602
603            // Verifica se os dígitos verificadores estão corretos
604            if ($cpfCnpj[9] == $digit1 && $cpfCnpj[10] == $digit2) {
605                return true;
606            } else {
607                return false;
608            }
609        }
610        // Verifica se é CNPJ
611        elseif (strlen($cpfCnpj) === 14) {
612            // Verifica se todos os dígitos são iguais
613            if (preg_match('/(\d)\1{13}/', $cpfCnpj)) {
614                return false;
615            }
616
617            // Calcula o primeiro dígito verificador
618            $sum = 0;
619            $weights = array(5, 4, 3, 2, 9, 8, 7, 6, 5, 4, 3, 2);
620            for ($i = 0; $i < 12; $i++) {
621                $sum += (int) ($cpfCnpj[$i]) * $weights[$i];
622            }
623            $digit1 = ($sum % 11 < 2) ? 0 : 11 - ($sum % 11);
624
625            // Calcula o segundo dígito verificador
626            $sum = 0;
627            $weights = array(6, 5, 4, 3, 2, 9, 8, 7, 6, 5, 4, 3, 2);
628            for ($i = 0; $i < 13; $i++) {
629                $sum += (int) ($cpfCnpj[$i]) * $weights[$i];
